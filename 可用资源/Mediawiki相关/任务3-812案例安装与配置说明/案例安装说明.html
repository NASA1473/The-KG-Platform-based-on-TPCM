window.onload = function () {
​
    var nav = document.getElementById("toc-id");
    var mark = document.getElementById("mark-point");
​
    scrollMenu(nav, mark);
​
}
​
function scrollMenu(obj, target) {
​
    window.onscroll = function () {
​
        var scrollTop = document.documentElement.scrollTop || document.body.scrollTop;
        var top = target.offsetTop;
        //当滚动高度大于目标元素的位置，导航条显示
        if (scrollTop >= top) {
            obj.className = "panel-body mw-collapsible-content toclimit-2 fixed-toc";
        } else {
            obj.className = "panel-body mw-collapsible-content toclimit-2";
        }
    }
}
